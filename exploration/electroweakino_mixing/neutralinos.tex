\section{Mixing matrix}

    \[
        M_{\tilde{\chi}^0} =
        \begin{bmatrix}
            M_1              & 0                & -m_Z c_\beta s_W & m_Z s_\beta s_W  \\
            0                & M_2              & m_Z c_\beta c_W  & -m_Z s_\beta c_W \\
            -m_Z c_\beta s_W & m_Z c_\beta c_W  & 0                & -\mu             \\
            m_Z s_\beta s_W  & -m_Z s_\beta c_W & -\mu             & 0
        \end{bmatrix}
    \]
